\begin{frame}
\frametitle{Enumerations}
\framesubtitle{Interlacing}

\begin{columns}
\begin{column}{.5\textwidth}
\begin{ttfamily}
\input{content/listings/itemize2_escaped.tex}
\end{ttfamily}
\end{column}
\begin{column}{.5\textwidth}
\begin{itemize}
\item first bullet item
\item second bullet item
\begin{itemize}
\item first subitem
\item second subitem
\end{itemize}
\item third bullet item
\item last bullet item
\end{itemize}
\end{column}
\end{columns}
\bigskip
In this way, subitems can be interlaced in maximal four layers.
\end{frame}
